ramroot=/dev/ram0 rw ramdisk_size=16384 initrd=0x84000000,16M
#mmcroot=/dev/mmcblk1p2 rw
ramrootfstype=ext2
#mmcrootfstype=ext4 rootwait
ramboot=setenv bootargs console=ttyO0,115200n8 root=${ramroot} rootfstype=${ramrootfstype}
#mmcboot=setenv bootargs console=ttyO0,115200n8 root=${mmcroot} rootfstype=${mmcrootfstype}
loadrd=fatload mmc ${mmcdev}:1 0x84000000 rd-ext2.bin;
loadzimage=fatload mmc ${mmcdev}:1 ${loadaddr} zImage;
loadfdt=fatload mmc ${mmcdev}:1 ${fdtaddr} am335x-boneblack.dtb;
uenvcmd=run loadrd; run loadzimage; run loadfdt; run ramboot; bootz ${loadaddr} - ${fdtaddr};
