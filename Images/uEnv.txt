mmcroot=/dev/mmcblk1p2 rw;
mmcrootfstype=ext4 rootwait;
mmcboot=setenv bootargs console=ttyO0,115200n8 root=${mmcroot} rootfstype=${mmcrootfstype}
loadzimage=fatload mmc ${mmcdev}:1 ${loadaddr} zImage;
#loadzimage=tftpboot ${loadaddr} zImage;
loadfdt=fatload mmc ${mmcdev}:1 ${fdtaddr} am335x-boneblack.dtb;
#loadfdt=tftpboot ${fdtaddr} am335x-boneblack.dtb;
uenvcmd=run loadzimage; run loadfdt; run mmcboot; bootz ${loadaddr} - ${fdtaddr};
